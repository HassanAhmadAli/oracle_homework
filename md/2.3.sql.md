CREATE OR REPLACE TRIGGER PreventTableOverlap BEFORE
INSERT ON ReservedTable FOR EACH ROW
DECLARE v_new_start_time DATE;
v_new_duration NUMBER;
v_new_end_time DATE;
v_conflict_count NUMBER;
BEGIN
SELECT TO_DATE(
        TO_CHAR(:NEW.ReservationDate, 'YYYY-MM-DD') || ' ' || r.ReservationTime,
        'YYYY-MM-DD HH24:MI'
    ),
    r.Period INTO v_new_start_time,
    v_new_duration
FROM Reservation r
WHERE r.CustomerID = :NEW.CustomerID
    AND r.ReservationDate = :NEW.ReservationDate;
v_new_end_time := v_new_start_time + (v_new_duration / 1440);
SELECT COUNT(*) INTO v_conflict_count
FROM ReservedTable rt
    JOIN Reservation r ON rt.CustomerID = r.CustomerID
    AND rt.ReservationDate = r.ReservationDate
WHERE rt.TableID = :NEW.TableID
    AND rt.ReservationDate = :NEW.ReservationDate
    AND (
        (
            TO_DATE(
                TO_CHAR(rt.ReservationDate, 'YYYY-MM-DD') || ' ' || r.ReservationTime,
                'YYYY-MM-DD HH24:MI'
            ) < v_new_end_time
        )
        AND (
            TO_DATE(
                TO_CHAR(rt.ReservationDate, 'YYYY-MM-DD') || ' ' || r.ReservationTime,
                'YYYY-MM-DD HH24:MI'
            ) + (r.Period / 1440) > v_new_start_time
        )
    );
IF v_conflict_count > 0 THEN RAISE_APPLICATION_ERROR(
    -20003,
    'Table is already reserved for this time period.'
);
END IF;
EXCEPTION
WHEN NO_DATA_FOUND THEN RAISE_APPLICATION_ERROR(-20004, 'Parent Reservation data not found.');
END;
/
