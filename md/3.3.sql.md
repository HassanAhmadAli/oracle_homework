CREATE MATERIALIZED VIEW LOG ON Customer WITH PRIMARY KEY;
CREATE MATERIALIZED VIEW LOG ON Reservation WITH PRIMARY KEY;
CREATE MATERIALIZED VIEW LOG ON ReservedTable WITH PRIMARY KEY;
CREATE MATERIALIZED VIEW LOG ON RestaurantTable WITH PRIMARY KEY;
CREATE MATERIALIZED VIEW LOG ON Location WITH PRIMARY KEY;
CREATE MATERIALIZED VIEW Customer_Reservation_Location_MV BUILD IMMEDIATE REFRESH FAST ON COMMIT AS
SELECT c.C_Name AS CustomerName,
    l.Name AS LocationName,
    COUNT(*) AS Reservation_Count
FROM Customer c
    JOIN Reservation r ON c.CustomerID = r.CustomerID
    JOIN ReservedTable rt ON r.CustomerID = rt.CustomerID
    AND r.ReservationDate = rt.ReservationDate
    JOIN RestaurantTable restbl ON rt.TableID = restbl.TableID
    JOIN Location l ON restbl.LocationID = l.LocationID
GROUP BY c.C_Name,
    l.Name;
/
